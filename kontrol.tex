\subsection{Конечные поля}


\begin{enumerate}
	\item Найти $(a_2x^2+a_1x+a_0)^{-1}$ в $GF(5)[x]/(x^3+x+1)$
	
	Если $i$ номер варианта, то последние три цифры $a_2a_1a_0$ в представлении числа $i+10$ в пятиричной системе счисления дают коэффициенты многочлена, т.е. $(a_2a_1a_0)_5=(i+10)_{10}$.
	
	\item Найти $(a_7x^7+a_6x^6+a_5x^5+a_4x^4+a_3x^3+a_2x^2+a_1x^1+a_0)^{-1}$ в $GF(2)[x]/(x^8+x^4+x^3+x+1)$

	Если $i$ номер варианта, то $a_7a_6a_5a_4a_3a_2a_1a_0$ есть двоичное представление числа $32\cdot i$.


\end{enumerate}


\subsection{Группы на эллиптических кривых}


\begin{enumerate}
	\item Построить график эллиптической кривой $y^2+xy=x^3+a_2x^2+a_6$ над полем $GF(2)[z]/(z^3+z+1)$ и сложить любые две непротивоположные точки. 
	
	Если $i$ номер варианта, то $a_2=b_2z^2+b_1z+b_0$, $a_6=с_2z^2+с_1z+с_0$, где $b_2b_1b_0$ есть двоичное представление числа $n=(i \pmod 7) +1$, а $c_2c_1c_0$  есть двоичное представление числа $(i-(i\pmod 7))/7+2$. 
	
	\item Построить график эллиптической кривой $y^2=x^3+ax^2+bx+c$ над полем $GF(3)[z]/(z^2+1)$ и сложить любые две непротивоположные точки. 

Если $i$ номер варианта, то $a=z$, $b=z+b_0$, $c=b_2z+b_1$, где $b_2b_1b_0$ есть троичное представление числа $i+27$. 	
	
	\item Построить график эллиптической кривой $y^2=x^3+ax+b$ над полем $GF(11)$ и сложить любые две непротивоположные точки. 
	
Если $i$ номер варианта, а $a_1a_0$ есть представление числа $i$ в десятичной системе счисления, то $a=a_0+1$, $b=a_1+2$.	

\end{enumerate}
